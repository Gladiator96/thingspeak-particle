768  int writeFields(unsigned long channelNumber, const char * writeAPIKey)
769  {
770  String postMessage = String("");
771  bool fFirstItem = true;
772  for(size_t iField = 0; iField < 8; iField++)
773  {
774  if(this->nextWriteField[iField].length() > 0)
775  {
776  if(!fFirstItem)
777  {
778  postMessage = postMessage + String("&");
779  }
780  postMessage = postMessage + String("field") + String(iField + 1) + String("=") + this->nextWriteField[iField];
781  fFirstItem = false;
782  this->nextWriteField[iField] = "";
783  }
784  }
785 
786  if(!isnan(nextWriteLatitude))
787  {
788  if(!fFirstItem)
789  {
790  postMessage = postMessage + String("&");
791  }
792  postMessage = postMessage + String("lat=") + String(this->nextWriteLatitude);
793  fFirstItem = false;
794  this->nextWriteLatitude = NAN;
795  }
796 
797  if(!isnan(this->nextWriteLongitude))
798  {
799  if(!fFirstItem)
800  {
801  postMessage = postMessage + String("&");
802  }
803  postMessage = postMessage + String("long=") + String(this->nextWriteLongitude);
804  fFirstItem = false;
805  this->nextWriteLongitude = NAN;
806  }
807 
808 
809  if(!isnan(this->nextWriteElevation))
810  {
811  if(!fFirstItem)
812  {
813  postMessage = postMessage + String("&");
814  }
815  postMessage = postMessage + String("elevation=") + String(this->nextWriteElevation);
816  fFirstItem = false;
817  this->nextWriteElevation = NAN;
818  }
819 
820  if(fFirstItem)
821  {
822  // setField was not called before writeFields
823  return ERR_SETFIELD_NOT_CALLED;
824  }
825 
826  return writeRaw(channelNumber, postMessage, writeAPIKey);
827  };

871  int writeRaw(unsigned long channelNumber, String postMessage, const char * writeAPIKey)
872  {
873  #ifdef PRINT_DEBUG_MESSAGES
874  Serial.print("ts::writeRaw (channelNumber: "); Serial.print(channelNumber); Serial.print(" writeAPIKey: "); Serial.print(writeAPIKey); Serial.print(" postMessage: \""); Serial.print(postMessage); Serial.println("\")");
875  #endif
876 
877  if(!connectThingSpeak())
878  {
879  // Failed to connect to ThingSpeak
880  return ERR_CONNECT_FAILED;
881  }
882 
883  postMessage = postMessage + String("&headers=false");
884 
885  #ifdef PRINT_DEBUG_MESSAGES
886  #ifdef SPARK
887  Particle.publish(SPARK_PUBLISH_TOPIC, "Post " + postMessage, SPARK_PUBLISH_TTL, PRIVATE);
888  #else
889  Serial.print(" POST \"");Serial.print(postMessage);Serial.println("\"");
890  #endif
891  #endif
892 
893  postMessage = postMessage + String("\n");
894 
895  // Post data to thingspeak
896  if(!this->client->print("POST /update HTTP/1.1\n")) return abortWriteRaw();
897  if(!writeHTTPHeader(writeAPIKey)) return abortWriteRaw();
898  if(!this->client->print("Content-Type: application/x-www-form-urlencoded\n")) return abortWriteRaw();
899  if(!this->client->print("Content-Length: ")) return abortWriteRaw();
900  if(!this->client->print(postMessage.length())) return abortWriteRaw();
901  if(!this->client->print("\n\n")) return abortWriteRaw();
902  if(!this->client->print(postMessage)) return abortWriteRaw();
903 
904  String entryIDText = String();
905  int status = getHTTPResponse(entryIDText);
906  if(status != OK_SUCCESS)
907  {
908  client->stop();
909  return status;
910  }
911  long entryID = entryIDText.toInt();
912 
913  #ifdef PRINT_DEBUG_MESSAGES
914  Serial.print(" Entry ID \"");Serial.print(entryIDText);Serial.print("\" (");Serial.print(entryID);Serial.println(")");
915  #endif
916 
917  client->stop();
918 
919  #ifdef PRINT_DEBUG_MESSAGES
920  Serial.println("disconnected.");
921  #endif
922  if(entryID == 0)
923  {
924  // ThingSpeak did not accept the write
925  status = ERR_NOT_INSERTED;
926  }
927  return status;
928  };

1301  bool connectThingSpeak()
1302  {
1303  bool connectSuccess = false;
1304  if(this->customIP == INADDR_NONE && NULL == this->customHostName)
1305  {
1306  #ifdef PRINT_DEBUG_MESSAGES
1307  Serial.print(" Connect to default ThingSpeak URL...");
1308  #endif
1309  connectSuccess = client->connect(THINGSPEAK_URL,THINGSPEAK_PORT_NUMBER);
1310  if(!connectSuccess)
1311  {
1312  #ifdef PRINT_DEBUG_MESSAGES
1313  Serial.print("Failed. Try default IP...");
1314  #endif
1315  connectSuccess = client->connect(THINGSPEAK_IPADDRESS,THINGSPEAK_PORT_NUMBER);
1316  }
1317  }
1318  else
1319  {
1320  if(!(this->customIP == INADDR_NONE))
1321  {
1322  // Connect to the server on port 80 (HTTP) at the customIP address
1323  #ifdef PRINT_DEBUG_MESSAGES
1324  Serial.print(" Connect to ");Serial.print(this->customIP);Serial.print("...");
1325  #endif
1326  connectSuccess = client->connect(this->customIP,this->port);
1327  }
1328  if(NULL != this->customHostName)
1329  {
1330  // Connect to the server on port 80 (HTTP) at the URL address
1331  #ifdef PRINT_DEBUG_MESSAGES
1332  #ifdef SPARK
1333  Particle.publish(SPARK_PUBLISH_TOPIC, "Attempt Connect to URL " + String(this->customHostName), SPARK_PUBLISH_TTL, PRIVATE);
1334  #else
1335  Serial.print(" Connect to ");Serial.print(this->customHostName);Serial.print(" ...");
1336  #endif
1337  #endif
1338  connectSuccess = client->connect(customHostName,this->port);
1339  }
1340  }
1341 
1342  #ifdef PRINT_DEBUG_MESSAGES
1343  if (connectSuccess)
1344  {
1345  #ifdef SPARK
1346  Particle.publish(SPARK_PUBLISH_TOPIC, "Connection Success", SPARK_PUBLISH_TTL, PRIVATE);
1347  #else
1348  Serial.println("Success.");
